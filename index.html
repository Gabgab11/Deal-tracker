<script>

let closerTotal = 0;

let openerTotal = 0;

let pendingDeal = {};


function login(){

  const role = document.getElementById('role').value;


  document.getElementById('loginPage').classList.add('hidden');


  if(role === 'opener'){

    document.getElementById('openerDashboard').classList.remove('hidden');

  }


  if(role === 'closer'){

    document.getElementById('closerDashboard').classList.remove('hidden');

  }


  if(role === 'ceo'){

    document.getElementById('ceoDashboard').classList.remove('hidden');

  }

}


function calculateCommission(){

  const amount = parseInt(document.getElementById('debtAmount').value) || 0;


  let commission = 0;


  if(amount >= 150 && amount < 200){

    commission = 10;

  }

  else if(amount >= 200 && amount < 251){

    commission = 25;

  }

  else if(amount >= 251 && amount < 301){

    commission = 40;

  }

  else if(amount >= 301 && amount < 401){

    commission = 50;

  }

  else if(amount >= 401 && amount < 500){

    commission = 60;

  }

  else if(amount >= 500){

    commission = 100;

  }


  document.getElementById('commission').value = commission;

}


function selectTransfer(type){

  document.getElementById('transferType').value = type;

}


function submitTransfer(){

  const type = document.getElementById('transferType').value;


  let commission = 0;


  if(type === 'Sales') commission = 5;

  if(type === 'Tax Above 10K') commission = 5;

  if(type === 'Tax Below 10K') commission = 2;

  if(type === 'Student Loan') commission = 2;


  openerTotal += commission;


  document.getElementById('openerCommission').innerText = '$' + openerTotal;


  const box = document.getElementById('transferStatus');

  box.classList.remove('hidden');

  box.innerHTML = 'Transfer Submitted Successfully. Pending CEO Review.';

}


function submitDeal(){

  const closer = document.getElementById('closerName').value;

  const client = document.getElementById('clientName').value;

  const amount = document.getElementById('debtAmount').value;

  const commission = document.getElementById('commission').value;


  pendingDeal = {

    closer,

    client,

    amount,

    commission

  };


  document.getElementById('dealStatus').classList.remove('hidden');

  document.getElementById('dealStatus').innerHTML = 'Deal Submitted. Waiting For CEO Approval.';


  document.getElementById('ceoCloser').innerText = closer;

  document.getElementById('ceoClient').innerText = client;

  document.getElementById('ceoDebt').innerText = amount;

  document.getElementById('ceoCommissionEdit').value = commission;

}


function approveDeal(){

  const editedCommission = parseInt(document.getElementById('ceoCommissionEdit').value) || 0;


  closerTotal += editedCommission;


  document.getElementById('closerEarnings').innerText = '$' + closerTotal;


  const result = document.getElementById('approvalResult');

  result.classList.remove('hidden');

  result.className = 'green';

  result.innerHTML = 'DEAL APPROVED SUCCESSFULLY';


  document.getElementById('dealStatus').className = 'green';

  document.getElementById('dealStatus').innerHTML = 'CEO APPROVED THIS DEAL';

}


function declineDeal(){

  const result = document.getElementById('approvalResult');

  result.classList.remove('hidden');

  result.className = 'red';

  result.innerHTML = 'DEAL DECLINED';


  document.getElementById('dealStatus').className = 'red';

  document.getElementById('dealStatus').innerHTML = 'CEO DECLINED THIS DEAL';

}
